indices = sample(1:nrow(iris), round(nrow(iris) * 0.8)) %>% sort()
train = iris[indices,]
test = iris[-indices,]
xtest = test[,-5]
ytest = test[, 5]

Fit neural network.

nnet = neuralnet(
  Species ~ .,
  train,
  hidden = c(4, 3),
  act.fct = "tanh",
  linear.output = FALSE
)
plot(nnet, rep = "best")

Get predictions.

ypred = neuralnet::compute(nnet, xtest)
yhat = ypred$net.result
print(yhat)
